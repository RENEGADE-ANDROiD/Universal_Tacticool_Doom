OptionMenu "WeaponTiltOptions" 
{
	Title "$UWT_TITL"
    StaticText "Universal Lean, Weapon Lowering & Rotational Tilting", gold
   
    StaticText ""

	Slider "$UWT_VELOCITY", "wt_rollvelocity", 0.50, 2.0, 0.01, 2

	Slider "$UWT_RESISTANCE", "wt_rollresistance", 0.10, 0.25, 0.01, 2

    StaticText ""

    Slider "$UWT_LOWERING_INTENSITY", "wt_lowering_intensity", 0.0, 2.0, 0.05, 2
    StaticText "0 = No lowering, 1 = Default, 2 = Double", "Teal"
    StaticText "Controls how much the weapon lowers when strafing.", "Teal"
    Option "Enable Weapon Lowering", "wt_lowering", "OnOff"
    StaticText ""

    Option "$UWT_CAP", "wt_cap", "OnOff"

    Slider "$UWT_CAPAMOUNT", "wt_rollcap", 1, 5, 0.5, 2

    StaticText ""
    
    Option "$UWT_DEBUG", "wt_debug", "OnOff"
    Option "$UWT_OFFSET", "wt_offset", "YesNo"

    StaticText ""
    StaticText "Stability Controls", "Gold"
    Option "Only Tilt While Moving", "wt_tilt_moving", "OnOff"
    Slider  "Minimum Movement Speed", "wt_tilt_moving_min", 0.0, 8.0, 0.1, 1
    Option "Only Tilt in Ready/Hold States", "wt_tilt_ready", "OnOff"
    StaticText "Helps reduce visual jitter during fire/reload animations.", "Teal"
    StaticText "Turn these on if another weapon mod has heavy offsets.", "Teal"

}

AddOptionMenu "OptionsMenu"
{
	Submenu "$UWT_TITL", "WeaponTiltOptions"
	StaticText ""
}

AddOptionMenu "OptionsMenuSimple"
{
	Submenu "$UWT_TIT2", "WeaponTiltOptions"
    StaticText ""
}
